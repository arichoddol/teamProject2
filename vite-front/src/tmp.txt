  const sendReplyData = async (boardIdValue, contentValue, memberIdValue )=>{ 

    const formData = new FormData();
    formData.append('content', contentValue);
    formData.append('memberId', memberIdValue)
    formData.append('boardId', boardIdValue);

    try{
   
      // Data get
      const url = `http://localhost:8088/api/reply/addReply?boardId=${boardIdValue}`;
      const response = await axios.post(url, formData);
      const saveReplyId = response.data;

      if(response.data && response.data.content){
          setContent(response.data.content);
      }
        console.log(saveReplyId);
        alert('댓글이 등록되었습니다');

        return saveReplyId;

    } catch(error){
      if (error.response) {
            // 서버가 응답을 반환했지만, 상태 코드가 2xx 범위 밖에 있을 때 (예: 400, 404, 500)
            console.error('댓글 등록 실패 (Server Error):', error.response.data);

            // 백엔드 서비스 로직 (게시글 ID 체크)에 따라 400 Bad Request가 올 수 있습니다.
            if (error.response.status === 400) {
                 alert('게시글 정보가 유효하지 않아 댓글을 등록할 수 없습니다.');
            } else {
                 alert('댓글 등록 중 오류가 발생했습니다.');
            }
        } else if (error.request) {
            // 요청은 전송되었으나 응답을 받지 못한 경우 (예: 네트워크 오류)
            console.error('댓글 등록 실패 (No Response):', '서버에 연결할 수 없습니다.');
        } else {
            // 요청을 설정하는 과정에서 오류가 발생한 경우
            console.error('Axios Error:', error.message);
        }
        throw error; // 에러를 다시 throw하여 호출한에서도 에러 처리가 가능하게 합니다.
    }
  }
